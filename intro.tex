function exam5_3_1
t=3:3:30;
G1=[10.2 13.9 12.16 13.49 13.74 12.01 11.55 11.02 12 12.12];
G2=[6.91 8.94 8.69 10.09 10.63 9.58 9.53 9.29 10.13 10.24];
A=1.83;
X=(G1-G2)./G2;
tt=linspace(3,30);
% 插值
pp=spline(t,X);
ppv=-fnval(pp,tt)/A;
% 微分
dp=fnder(pp);dpv=fnval(dp,tt);
plot(t,X,'ro',tt,ppv,tt,dpv,'.-')
axisX=refline(0,0)
axisX.Color='c'
legend('原始数据','拟合曲线','导数曲线','导数参考直线')
\end{lstlisting}
